#Question 7: Lấy ra ID của question có >=4 câu trả lời
select
	QuestionA.QuestionID
from
	(
	select
		QuestionID,
		count(*) as SL
	from
		answer
	group by
		QuestionID
	having
		SL >= 4) as QuestionA ;
